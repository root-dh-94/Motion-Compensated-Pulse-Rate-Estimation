import glob
import scipy.signal
import numpy as np
import scipy as sp
import scipy.io
import matplotlib.pyplot as plt

def LoadTroikaDataset():
    """
    Retrieve the .mat filenames for the troika dataset.

    Review the README in ./datasets/troika/ to understand
    the organization of the .mat files.

    Returns:
        data_fls: Names of the .mat files that contain signal data
        ref_fls: Names of the .mat files that contain reference data
        <data_fls> and <ref_fls> are ordered correspondingly,
        so that ref_fls[5] is the
        reference data for data_fls[5], etc...
    """
    data_dir = "./datasets/troika/training_data"
    data_fls = sorted(glob.glob(data_dir + "/DATA_*.mat"))
    ref_fls = sorted(glob.glob(data_dir + "/REF_*.mat"))
    return data_fls, ref_fls


def LoadTroikaDataFile(data_fl):
    """
    Loads and extracts signals from a troika data file.

    Usage:
        data_fls, ref_fls = LoadTroikaDataset()
        ppg, accx, accy, accz = LoadTroikaDataFile(data_fls[0])

    Args:
        data_fl: (str) filepath to a troika .mat file.

    Returns:
        numpy arrays for ppg, accx, accy, accz signals.
    """
    data = sp.io.loadmat(data_fl)['sig']
    return data[2:]


def Fourier(signal, fs=125):
    """
    Computes fft for given signal

    Args:
        signal: given signal for which we want fft

    Returns:
        array of frequencies and fft for the signal
    """
    normalise_signal = (signal - np.mean(signal))/(max(signal)-min(signal))
    padded = 4*len(normalise_signal)
    freqs = np.fft.rfftfreq(padded, 1/fs)
    fft = np.fft.rfft(normalise_signal, padded)
    return freqs, fft


def BandpassFilter(s, pass_band=(40/60, 240/60), fs=125):
    """
    Attenuates frequencies outside passband

    Args:
        signal: given signal that is filtered
        pass_band:  range of frequencies in signal that are not attenuated

    Returns:
        signal with frequencies outside passband attenuated
    """
    b, a = sp.signal.butter(3, pass_band, btype='bandpass', fs=fs)
    return sp.signal.filtfilt(b, a, s)


def AggregateErrorMetric(pr_errors, confidence_est):
    """
    Computes an aggregate error metric based on confidence estimates.

    Computes the MAE at 90% availability.

    Args:
        pr_errors: a numpy array of errors between
        pulse rate estimates and corresponding
        reference heart rates.
        confidence_est: a numpy array of confidence
        estimates for each pulse rate
        error.

    Returns:
        the MAE at 90% availability
    """
    # Higher confidence means a better estimate. The best 90% of the estimates
    #    are above the 10th percentile confidence.
    percentile90_confidence = np.percentile(confidence_est, 10)

    # Find the errors of the best pulse rate estimates
    best_estimates = pr_errors[confidence_est >= percentile90_confidence]

    # Return the mean absolute error
    return np.mean(np.abs(best_estimates))


def Evaluate():
    """
    Top-level function evaluation function.

    Runs the pulse rate algorithm on the Troika dataset
    and returns an aggregate error metric.

    Returns:
        Pulse rate error on the Troika dataset. See AggregateErrorMetric.
    """
    # Retrieve dataset files
    data_fls, ref_fls = LoadTroikaDataset()
    errs, confs = [], []
    for data_fl, ref_fl in zip(data_fls, ref_fls):
        # Run the pulse rate algorithm on each trial in the dataset
        errors, confidence = RunPulseRateAlgorithm(data_fl, ref_fl)
        errs.append(errors)
        confs.append(confidence)
        # Compute aggregate error metric
    errs = np.hstack(errs)
    confs = np.hstack(confs)
    return AggregateErrorMetric(errs, confs)


def RunPulseRateAlgorithm(data_fl, ref_fl):
    """
    Runs analysis on each 8s window of a given
    signal to predict Pulse frequency.
    Compiles error in this frequency as well as
    confidence in our predction.

    Args:
        signal: given signal for which we want to determine pulse
        ref_f1: reference data of pulse against which
        we compare our prediction

    Returns:
        tuple of arrays.The error aray consists of the
        difference between our prediction
        and the reference pulse rate for each 8s window.
        The confidence array contains
        level of confidence in predictions
    """
    # Load data using LoadTroikaDataFile
    ppg, accx, accy, accz = LoadTroikaDataFile(data_fl)

    # Load reference HR
    ref = sp.io.loadmat(ref_fl)['BPM0']

    # sampling rate found to be 125Hz(from Readme)
    fs = 125

    # 8s windows with a 2s shift
    wind_len = 8*fs
    wind_shift = 2*fs
    start = 0

    error, confidence = [], []
    # Initialize previous pulse estimate to compare next prediction with
    prev_est = None
    # exploring each 8s window, every 2s
    for _ in range(len(ref)):
        ppg_wind = ppg[start:(start+wind_len)]
        accx_wind = accx[start:(start+wind_len)]
        accy_wind = accy[start:(start+wind_len)]
        accz_wind = accz[start:(start+wind_len)]
        start = start + wind_shift

        # Call the 'Compute' fnc to get prediction and confidence
        conf, pred = Compute(ppg_wind, accx_wind, accy_wind,
                             accz_wind, prev_est)

        # reference heartbeat for this window
        ref_bpm = ref[_]

        err = ref_bpm - pred
        error.append(err)
        confidence.append(conf)
        prev_est = pred

    # Return per-estimate error and confidence as a 2-tuple of numpy arrays.
    errors, confidence = np.array(error), np.array(confidence)
    errors = errors.reshape(errors.shape[0],)

    return (errors, confidence)


def Compute(ppg_wind, accx_wind, accy_wind, accz_wind, prev_est, fs=125):
    """
    Runs the bulk of detailed analysis on our signal
    for a given time window,
    to compute and predict pule rate and confidence
    for the window

    Args:
        ppg_wind: ppg signal for which we want to determine pulse rate
        accx_wind, accy_wind, accz_wind: channels from the
        accelerometer that are used to determine the correct
        frequencies concerning the ppg signal
        prev_est: estimate of pulse rate in signal, from the
        previous window
        fs: sampling rate

    Returns:
        prediction of pulse rate for this window, along with the
        confidence in that prediction
    """

    # bandpass ppg and acc signal, only including freq between 40 BPM/240BPM
    filtered_ppg = BandpassFilter(ppg_wind)
    # filtered_acc = BandpassFilter(np.sqrt
    #                              (accx_wind**2+accy_wind**2+accz_wind**2))
    filtered_accx = BandpassFilter(accx_wind)
    filtered_accy = BandpassFilter(accy_wind)
    filtered_accz = BandpassFilter(accz_wind)

    # FFT on ppg and acc channels
    freq_ppg, fft_ppg = Fourier(filtered_ppg)
    # freq_acc, fft_acc = Fourier(filtered_acc)
    freq_accx, fft_accx = Fourier(filtered_accx)
    freq_accy, fft_accy = Fourier(filtered_accy)
    freq_accz, fft_accz = Fourier(filtered_accz)

    low = 40/60
    high = 240/60

    # get ride of unwanted frequencies
    freq_ppg_filt = freq_ppg[(freq_ppg >= low) & (freq_ppg < high)]
    # freq_acc_filt = freq_acc[(freq_acc >= low) & (freq_acc < high)]
    freq_accx_filt = freq_accx[(freq_accx >= low) & (freq_accx < high)]
    freq_accy_filt = freq_accy[(freq_accy >= low) & (freq_accy < high)]
    freq_accz_filt = freq_accz[(freq_accz >= low) & (freq_accz < high)]

    # get magnitude of frequencies of interest
    mag_ppg = np.abs(fft_ppg)[(freq_ppg >= low) & (freq_ppg < high)]
    # mag_acc = np.abs(fft_acc)[(freq_acc >= low) & (freq_acc < high)]
    mag_accx = np.abs(fft_accx)[((freq_accx >= low) & (freq_accx < high))]
    mag_accy = np.abs(fft_accy)[((freq_accy >= low) & (freq_accy < high))]
    mag_accz = np.abs(fft_accz)[((freq_accz >= low) & (freq_accz < high))]

    # find highest peaks
    peaks_ppg = sp.signal.find_peaks(mag_ppg, height=10, distance=1)[0]
    # peaks_acc = sp.signal.find_peaks(mag_acc, height=15, distance=1)[0]
    peaks_accx = sp.signal.find_peaks(mag_accx, height=10, distance=1)[0]
    peaks_accy = sp.signal.find_peaks(mag_accy, height=10, distance=1)[0]
    peaks_accz = sp.signal.find_peaks(mag_accz, height=10, distance=1)[0]

    # sort peaks from highest to lowest and select 3 highest
    sort_ppg = sorted(peaks_ppg, key=lambda i: mag_ppg[i], reverse=True)[0:3]
    # sort_acc = sorted(peaks_acc, key=lambda i: mag_acc[i], reverse=True)[0:3]
    sort_acx = sorted(peaks_accx, key=lambda i: mag_accx[i], reverse=True)[0:3]
    sort_acy = sorted(peaks_accy, key=lambda i: mag_accy[i], reverse=True)[0:3]
    sort_acz = sorted(peaks_accz, key=lambda i: mag_accz[i], reverse=True)[0:3]

    closest_to_prev = []
    select_peak = None
    peak_list = []
    stm = 60
    # iterate over list of 3 highst peaks in ppg fft
    for peak in sort_ppg:

        # if a previous prediction for pulse rate is avalaible,
        # find the difference between the estimated pulse
        # rate if using the current peak frequency,and the
        # previous prediction. Add the differences to an array
        if prev_est is not None:
            closest_to_prev.append(np.abs((freq_ppg_filt[peak]*stm)-prev_est))

        # Check if current peak is also found in the acc channel signals
        if ((peak in sort_acx) or (peak in sort_acy) or (peak in sort_acz)):

            # remove this current iteration since this frequency is
            # probably due to some movement picked up by the accelerometer
            try:
                closest_to_prev.remove(np.abs((freq_ppg_filt
                                               [peak]*stm)-prev_est))
            except:
                True

        else:
            peak_list.append(peak)

    # Choose frequency that is closest to the previous pulse rate
    # prediction, since we expect pulse rate to change gradually
    # (not available for very first window)
    if prev_est is not None and closest_to_prev:
        select_peak = peak_list[(closest_to_prev.index(min(closest_to_prev)))]

    # This is to choose frequency of interest for very first window
    # since there will not be a prev_est
    elif peak_list:
        select_peak = peak_list[0]

    # If all the peaks in the ppg fft are also found in the acc fft,
    # choose the first highest peak since in this case it is likely
    # that the frequency is the same for the ppg and acc
    else:
        select_peak = sort_ppg[0]

    # select pulse freq
    pred = freq_ppg_filt[select_peak]

    # select BPM
    bmp = pred*stm

    confidence = ConfEst(freq_ppg, fft_ppg, mag_ppg, pred)
    return confidence, bmp


def ConfEst(freq_ppg, fft_ppg, mag_ppg, pred):
    """
    Computes confidence in our prediction by measuring the proportion of
    energy found around our frequency when compared to the energy across
    all frequencies

    Args:
        freq_ppg: range of frequencies in our signal
        fft_ppg: fft of our ppg signal
        mag_ppg: magnitude of frequencies
        pred: our prediction for pulse rate

    Returns:
        proportion of energy found around our frequency when compared
        to the energy across all frequencies, as confidence
    """
    wind = 0.5
    freq_window = (freq_ppg >= (pred - wind)) & (freq_ppg <= (pred + wind))
    confidence = np.sum(np.square(np.abs(fft_ppg)
                                  [freq_window]))/np.sum(np.square(mag_ppg))

    return confidence

Project Write-up¶

Code Description The algorithm is used to predict the pulse rate from a PPG signal and a 3-axis accelerometer signal, and outputs the Mean Absolute Error when compared to a datset containing a reference pule rate signal. The algorithm only functions for Pulse Rates between 40BMP and 240 BPM.

The algorithm consists of a top level function 'Evaluate' that needs to be called. This function is called to receive the Mean absolute Error of our predictions across the TROIKA dataset. The Evaluate function compiles the list of data files and the reference data files from each subject and feeds the data and the reference data file for each subject individually, to the RunPulseRateAlgorithm.

The RunPulseRateAlgorithm fnc can be called specifically to receive the actual pulse rate predictions every 2 seconds, along with the confidence in those predictions. This functions expects 2 '.mat' files, one being the signal on which we want to determine the pulse rate, and one being a reference pulse rate dataset.

The model predicts the pulse rate by first passing a specific window of PPG and accelerometer signals through a Bandpass filter, follow by a FFT to determine the dominating frequencies in each signal. Hereafter, the top 3 dominating frequencies for each signal are selected, and the ones that occur in both the PPG signal, and the accelerometer signal are discarded as these are probably caused by motion of the subject. We then choose the frequency that would provide us with a pulse estimate most similar to our previous time window estimate. The confidence of our estimate is based off the relative spectral energy in our PPG signal around our selected frequency, when compared to the total spectral energy across all frequencies.

Furthermore, an error is calculated by each estimated by subtracting it from our reference pulse rate for that particular time window. The Evaluate function compiles all our errors and confidences across all windows in a dataset, and across all datsets, to finally output a mean absolute error.

The code also uses the following libraries 1)glob 2)scipy.signal 3)numpy - version 1.12.1 4)scipy - version 1.2.1 5)scipy.io 6)matplotlib.pyplot 7)scipy.signal.find_peaks

Data Description The dataset comprise the TROIKA dataset. The data from 12 subjects aged 18-35. Data was collected from subjects while the were at rest, and while they were running on the treadmill at various speeds. The subjects were divided into two groups, where one group ran at a consistently higher pace. Each subject contributed two different files for the data. The first file comprised 6 rows divided into ECG, PPG and 3-axis accelerometer data, of which we only used the last 4 rows, namely the PPG and 3-axis accelerometer data. Additionaly, the other file for each subject had the reference heart rate against which our predictions were validated. The signals was sampled at 125Hz. The dataset could be improved in the following ways:

1)More subjects included in the study, along with more datapoints, would help improve the dataset. Subjects tend to have homogeneity within their data and therefore, data from diverse subjects will help our algorithm.

2)Data from more activities. It would help to gather data on subjects while they are undergoing a wider range of activities, such as biking or possibly weight training. This will also further vary our data and help our algorithm generalise better.

3)It would be beneficial and provide more accurate results if the PPG sensor was not mounted on the wrist, but instead a more stationary part of the body that would not result in the accelerometer signal inducing as much noise into the PPG signal.

Algorithm Description

How it works:

The algorithm is called through its top level function 'Evaluate'. This functions is used to run the 'RunPulseRateAlgorithm' function on the TROIKA dataset. The 'RunPulseRateAlgorithm' function accepts the file containing the PPF and 3-axis accelerometer signal as well as the file containing the reference Pulse rate, as arguments. The analysis of the PPG signal is broken down into 8s windows and a predicitons for Pulse rate along with confidence is output every 2s. Therefore, each new window has a 6s overlap with the previous window. After loading in the PPG and accelerometer signals, as mentioned previously we loop over the signal in 8 s increments. Each of these 8s window increment signals are sent to the function 'Compute' for the main analysis. Inside 'Compute' the 8s signals are passed through a Bandpass filter to remove any frequencies below 40 BPM and above 240 BPM since these are physiologically unrealistic. We then carry out FFT on our signals to obtain the list of prominent frequencies and their magnitudes. We then sort these frequencies and choose the 3 highest frequency peaks for each signal my calling the 'find_peaks' function. Now our aim is to figure out which frequencies are noise contributed by movement and which frequencies indeed represent the pulse rate. We do this my comparing the peaks of the PPG signal and the accelerometer signals. If a peak in the PPG signal is also found in one of the acceloremeter signal, it is more than likely that this frequency corresponding to this peak is just noise and not the pulse rate frequency we are after. From the remaining peaks, we choose the one that would provide us a pulse rate estimation that is most similar to the one from the window before the current window being analysed. We do thos because it is unlikely that there will be big differences in pulse rate from one window to another. However, if we find that all the peaks that occur in the PPG signal also occur in the accelerometer signal, it is probable that the pulse rate frequency is the same as that picked up by the accelerometer. In this case we just choose the highes PPG signal peak. Once we have selected our peak, we determine the freqeuncy that corresponds to this peak and multiply that by 60 to get the Pulse rate. We then calculate the confidence in our prediction by computing the spectral energy around our selected frequency and then dividing that by the spectral energy across all frequncyies. We return our prediction of pulse rate, along with the confidence to the 'RunPulseRateAlgorithm' function. Now we compare our prediction to the reference pulse rate for the first window. We calculate the error by subtracting the two. Through an array of these errors and confidence, for each window, in each file, we finally obtain a Mean Absolute Error for our predictions, which is finally output by our 'Evaluate' function.

The specific aspects of the physiology that it takes advantage of:

The signal received by the algorithm is obtained by a PPG sensor. This sesnor shines a light, penetrating our skin and reflecting of the capillaries under our skin. The amount of reflected light is then picked up by a photodiode that in turn generates a voltage depending on the amount of reflected light. When the ventricles contract, blood enters the capillaries and a majority of the light is absorbed by the blood, with a minimal amount being reflected. When the ventricles are rleaxed and blood leaves the caplillaries, a large amount of light is relfected. This cycle generates a signal in the PPG which we then process to determin the Pulse Rate.

A describtion of the algorithm outputs:

The 'RunPulseRateAlgorithm' outputs our prediction for the Pulse rate for that specific 8s window, along with a confidence level in our prediction, based on th relative spectral energy. The 'Evaluate' function outputs the Mean Absolute Error of our predictions, after running our algorithm on the entire dataset

Caveats on algorithm outputs:

Our predictions are made for 8s windows and assume that the heart rate is not changing drastically in an immediate manner. our confidence is also based on the signal to noise ratio and hence judges the relative signal energy around our selected frequency, rather than the accuracy of our output. Furthermore, we do not predict any Pulse rate below 40 BPM. It is also possible that the algorithm is optimised for this particular dataset and particular noise levels, and might have slightly lower performancy on a brand new dataset. We also note that our MAE is 9.9 BPM howeveer, for any specific prediction the actual error can be more than that.

Common failure modes:

1) Pulse rates less than 40 BPM or more than 240 BPM 2) Drastic changes in Pulse rates in a short time <2s 3) Dominating accelerometer signal frequencie and harmonics

Algorithm Performance

The model was validated against the reference Pulse rates in the TROIKA dataset. A Mean Absolute Error was computed against 90% availabilty of the Troika dataset(bottome 10% not considered). This resulted in a MAE of 9.9 BPM over the top 90% of predictions.

It is unlikely that the algorithm will generalize and match this performance when given a different dataset as it was optimized using this specific dataset. Subjects have some ingerent homegeneity in their data so it is possible that the algorithm would not perform as well when exposed to a completely different subject, or even data while the subject is undertaking a different excercise
